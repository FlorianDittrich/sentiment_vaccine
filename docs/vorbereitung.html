Wir schreiben eine Funktion, um den Text zu bereinigen. Twitter interne Kennzeichnungen, Punktion, Links etc. sollen entfernt werden

clean.text=function(x)
  {x=  gsub("rt", "",x) 
  x=gsub("@\\w+", "",x) 
  x=gsub("[[:punct:]]", "",x)  
  x=gsub("http\\w+", "",x)  
  x=gsub("^ ", "",x) 
  x=gsub(" $", "",x)
  return(x)
}

data$text<-clean.text(data$text)
